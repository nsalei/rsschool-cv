def df_cols_to_list(df, cols_l):
'''
Convert cols tp list with edits (upper, trim)
:param df: df
:param cols_l: list of cols
:return: list
'''
lst = []
for column in cols_l:
    for row in df[column].unique():
        lst.append(str(row).strip().upper())
return list(filter(None, lst))
